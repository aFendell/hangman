# Hangman Game
from unit2_opening_screen_V2 import opening_screen
from unit9_choose_word import choose_word
from unit8_print_hangman import print_hangman
from unit3_word_pattern import word_pattern
from unit6_check_valid_input import check_valid_input
from unit6_try_update_letter_guessed import try_update_letter_guessed
from unit7_show_hidden_word import show_hidden_word
from unit7_check_win import check_win


def main():
    # Present 'Hangman' opening screen
    opening_screen()

    # Ask user to enter file_path and an index number to select a word
    file_path = input("Please enter a path to a text file: ")
    index = input("Please enter an index number to select a word: ")
    # Choose a secret word
    secret_word = choose_word(file_path, int(index))

    # Present the 1st 'Hangman' image
    print_hangman(1)

    # Present the secret_word pattern
    word_pattern(secret_word)

    old_letters_guessed = []
    num_of_tries = 1
    # run a loop for max 6 failed attempts
    while num_of_tries < 7:
        # Ask user to guess a single letter and press enter
        ch = (input("Please guess a single alphabetic letter: ")).lower()
        # Check if character is valid - single english letter and not already in the old_letters_guessed list
        # If not valid or , print 'X' and the old letters guessed separated with arrows
        if not check_valid_input(ch, old_letters_guessed):
            try_update_letter_guessed(ch, old_letters_guessed)
            continue
        else:
            # If valid, update 'old_letters_guessed' list
            try_update_letter_guessed(ch, old_letters_guessed)
            # If not in secret_word - failed attempt - print next step of 'Hangman'
            if ch not in secret_word:
                num_of_tries += 1
                print(":(")
                print_hangman(num_of_tries)
            else:
                # Check for win - guessed all letters of secret_word
                if check_win(secret_word, old_letters_guessed):
                    show_hidden_word(secret_word, old_letters_guessed)
                    print("Win")
                    break
            # Print partial secret_word
            show_hidden_word(secret_word, old_letters_guessed)
    if num_of_tries == 7:
        print("\n    LOSE ! ! !\n    GAME OVER ! ! !")

if __name__ == "__main__":
    main()
